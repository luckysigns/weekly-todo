Sync — Not connected

This app auto-connects to your Supabase project on load using the key baked into the file, so you shouldn't need to touch this on a new device. Uses an open access policy on your two tables — fine for personal/solo use, just don't share the project URL/key. Disconnecting here just stops this browser from syncing; reload to reconnect.

-- Run once in your Supabase project's SQL editor

create table if not exists todo_tasks (
  id text primary key,
  title text not null,
  category text default '',
  date date,
  time text,
  loc text,
  deadline date,
  done boolean default false,
  updated_at timestamptz default now()
);

create table if not exists trip_status (
  id text primary key default 'main',
  currently_in text,
  currently_in_start date,
  currently_in_end date,
  heading_to text,
  heading_to_date date,
  updated_at timestamptz default now()
);

alter table todo_tasks enable row level security;
alter table trip_status enable row level security;

create policy "allow all todo_tasks" on todo_tasks for all using (true) with check (true);
create policy "allow all trip_status" on trip_status for all using (true) with check (true);

alter publication supabase_realtime add table todo_tasks;
alter publication supabase_realtime add table trip_status;
