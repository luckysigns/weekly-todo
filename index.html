Auto-connects using the key baked into this file. Open access policy on your tables — fine for personal use, just don't share the URL/key. If you already ran the setup SQL before, re-run it — it's additive and safe (adds the new categories table and date_end column).

-- Run once (or re-run — it's additive) in your Supabase project's SQL editor

create table if not exists todo_tasks (
  id text primary key,
  title text not null,
  category text default '',
  date date,
  date_end date,
  time text,
  loc text,
  deadline date,
  done boolean default false,
  updated_at timestamptz default now()
);
alter table todo_tasks add column if not exists date_end date;

create table if not exists trip_status (
  id text primary key default 'main',
  currently_in text,
  currently_in_start date,
  currently_in_end date,
  updated_at timestamptz default now()
);

create table if not exists destinations (
  id text primary key,
  location text not null,
  start_date date,
  end_date date,
  updated_at timestamptz default now()
);

create table if not exists categories (
  id text primary key,
  label text not null,
  color text not null default '#8a8579',
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

alter table todo_tasks enable row level security;
alter table trip_status enable row level security;
alter table destinations enable row level security;
alter table categories enable row level security;

drop policy if exists "allow all todo_tasks" on todo_tasks;
drop policy if exists "allow all trip_status" on trip_status;
drop policy if exists "allow all destinations" on destinations;
drop policy if exists "allow all categories" on categories;

create policy "allow all todo_tasks" on todo_tasks for all using (true) with check (true);
create policy "allow all trip_status" on trip_status for all using (true) with check (true);
create policy "allow all destinations" on destinations for all using (true) with check (true);
create policy "allow all categories" on categories for all using (true) with check (true);

alter publication supabase_realtime add table todo_tasks;
alter publication supabase_realtime add table trip_status;
alter publication supabase_realtime add table destinations;
alter publication supabase_realtime add table categories;
